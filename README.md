### Importing with Maven
```xml
<repositories>
    <repository>
        <id>label-repo</id>
        <url>https://raw.github.com/Lucaniuss/Label/repository/</url>
    </repository>
</repositories>
    
<dependencies>
    <dependency> <!-- Label -->
        <groupId>me.lucanius.label</groupId>
        <artifactId>Label</artifactId>
        <version>1.0</version>
        <scope>compile</scope>
    </dependency>
</dependencies>
```
